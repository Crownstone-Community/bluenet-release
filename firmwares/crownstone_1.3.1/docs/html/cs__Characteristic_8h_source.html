506  void written(uint16_t len) {
508  //TODO: have this from the stack directly.
509  if (EncryptionHandler::getInstance().allowedToWrite()) {
510  LOGi("Not allowed to write, disconnect in progress")
511  return;
512  }
513 
514  setGattValueLength(len);
515 
516  EncryptionAccessLevel accessLevel = NOT_SET;
517  // when using encryption, the packet needs to be decrypted
518  if (_status.aesEncrypted && _minAccessLevel < ENCRYPTION_DISABLED) {
519 
520  // the arithmetic type- and the string type characteristics have their hardcoded length in the valueLength
521  // the getValueLength() for those two types will always be the same and the setValueLength does nothing there.
522  // In the case of a characteristic with a dynamic buffer length we need to set the length ourselves.
523  // To do this we assume the length of the data is the same as the encrypted buffer minus the overhead for the encryption.
524  // The result can be zero padded but generally the payload has it's own length indication.
525  uint16_t decryptionBufferLength = EncryptionHandler::calculateDecryptionBufferLength(getGattValueLength());
526  setValueLength(decryptionBufferLength);
527 
528  bool success = EncryptionHandler::getInstance().decrypt(
529  getGattValuePtr(),
530  getGattValueLength(),
531  getValuePtr(),
532  getValueLength(),
533  accessLevel
534  );
535 
536  // disconnect on failure or if the user is not authenticated
537  if (!success || !EncryptionHandler::getInstance().allowAccess(_minAccessLevel , accessLevel)) {
538  LOGi("insufficient access")
539  EncryptionHandler::getInstance().closeConnectionAuthenticationFailure();
540  return;
541  }
542  }
543  else {
544  accessLevel = ENCRYPTION_DISABLED;
545  setValueLength(len);
546  }
547 
548 
549 
550  LOGd("%s: onWrite()", _name);
551 
552  _callbackOnWrite(accessLevel, getValue());
553  }

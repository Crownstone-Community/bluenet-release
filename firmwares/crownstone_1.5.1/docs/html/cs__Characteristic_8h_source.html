508  void written(uint16_t len) {
510  //TODO: have this from the stack directly.
511  if (EncryptionHandler::getInstance().allowedToWrite()) {
512  LOGi("Not allowed to write, disconnect in progress")
513  return;
514  }
515 
516  setGattValueLength(len);
517 
518  EncryptionAccessLevel accessLevel = NOT_SET;
519  // when using encryption, the packet needs to be decrypted
520  if (_status.aesEncrypted && _minAccessLevel < ENCRYPTION_DISABLED) {
521 
522  // the arithmetic type- and the string type characteristics have their hardcoded length in the valueLength
523  // the getValueLength() for those two types will always be the same and the setValueLength does nothing there.
524  // In the case of a characteristic with a dynamic buffer length we need to set the length ourselves.
525  // To do this we assume the length of the data is the same as the encrypted buffer minus the overhead for the encryption.
526  // The result can be zero padded but generally the payload has it's own length indication.
527  uint16_t decryptionBufferLength = EncryptionHandler::calculateDecryptionBufferLength(getGattValueLength());
528  setValueLength(decryptionBufferLength);
529 
530  bool success = EncryptionHandler::getInstance().decrypt(
531  getGattValuePtr(),
532  getGattValueLength(),
533  getValuePtr(),
534  getValueLength(),
535  accessLevel
536  );
537 
538  // disconnect on failure or if the user is not authenticated
539  if (!success || !EncryptionHandler::getInstance().allowAccess(_minAccessLevel , accessLevel)) {
540  LOGi("insufficient access")
541  EncryptionHandler::getInstance().closeConnectionAuthenticationFailure();
542  return;
543  }
544  }
545  else {
546  accessLevel = ENCRYPTION_DISABLED;
547  setValueLength(len);
